export default { async fetch(request) { const url = new URL(request.url) const target = url.searchParams.get("url") if (!target) { return new Response("No URL", { status: 400 }) } const headers = new Headers(request.headers) headers.set("Origin", "") const resp = await fetch(target, { headers, method: request.method }) const newHeaders = new Headers(resp.headers) newHeaders.set("Access-Control-Allow-Origin", "*") newHeaders.set("Access-Control-Allow-Headers", "*") newHeaders.set("Access-Control-Allow-Methods", "GET,HEAD,OPTIONS") newHeaders.set("Content-Type", resp.headers.get("Content-Type") || "application/vnd.apple.mpegurl") return new Response(resp.body, { status: resp.status, headers: newHeaders }) } }
